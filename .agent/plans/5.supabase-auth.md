# 5. Supabase Auth

⚠️ **Medium** — Touches frontend, backend, and database. Validate piece-by-piece.

## Context

The frontend `login.tsx` is a styled form that calls `setUser({email})` after a 650ms timeout (lines 21–27). `app-shell.tsx` gates the whole app on `!user` (line 407) and hydrates `projects` from `fixtures.ts` (line 38). Sub-plan 4 stood up a FastAPI scaffold but it has no auth and no DB. This sub-plan delivers real auth: Supabase manages users + sessions, Postgres stores projects/chats/project_files with RLS, and the FastAPI backend validates Supabase JWTs on every request.

## Goal

A new user can sign up via the existing login form, lands in the app shell, sees a default project seeded server-side, and a second user cannot see the first user's data via any direct REST call.

## Prerequisites (user-provided)

- Supabase Cloud project provisioned. Have ready: `SUPABASE_URL`, anon key, service-role key, JWT secret (Project Settings → API).
- Email/password auth enabled in Supabase Auth settings; "Confirm email" can stay on or off — if on, dev-confirm via dashboard or use a temp email.

Populate the **single root `.env`** (copy from `.env.example`):
```
NEXT_PUBLIC_SUPABASE_URL=<from dashboard>
NEXT_PUBLIC_SUPABASE_ANON_KEY=<anon public key from dashboard>
NEXT_PUBLIC_API_URL=http://localhost:8000

SUPABASE_URL=<same URL>
SUPABASE_SERVICE_ROLE_KEY=<service-role key>
SUPABASE_JWT_SECRET=<JWT secret from API settings>
```
Both Next.js and FastAPI read from this one file (FastAPI via absolute path resolved in `backend/app/config.py`).

### 5.1 Database schema migration

Create `supabase/migrations/0001_init.sql`:

```sql
create extension if not exists "pgcrypto";

create table public.projects (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references auth.users(id) on delete cascade,
  name text not null,
  openai_vector_store_id text,
  created_at timestamptz default now()
);

create table public.chats (
  id uuid primary key default gen_random_uuid(),
  project_id uuid not null references public.projects(id) on delete cascade,
  user_id uuid not null references auth.users(id) on delete cascade,
  title text not null,
  openai_thread_id text,
  created_at timestamptz default now()
);

create table public.project_files (
  id uuid primary key default gen_random_uuid(),
  project_id uuid not null references public.projects(id) on delete cascade,
  user_id uuid not null references auth.users(id) on delete cascade,
  filename text not null,
  size_bytes bigint,
  openai_file_id text,
  status text default 'pending',
  created_at timestamptz default now()
);

alter table public.projects enable row level security;
alter table public.chats enable row level security;
alter table public.project_files enable row level security;

create policy "owner full access" on public.projects
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);
create policy "owner full access" on public.chats
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);
create policy "owner full access" on public.project_files
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);

-- Seed a default project on first signup
create or replace function public.handle_new_user()
returns trigger language plpgsql security definer set search_path = public as $$
begin
  insert into public.projects (user_id, name) values (new.id, 'My Project');
  return new;
end;
$$;

create trigger on_auth_user_created
  after insert on auth.users
  for each row execute function public.handle_new_user();
```

**Apply via Supabase Dashboard → SQL editor (paste + run), or via `supabase` CLI if installed locally.**

**Validation:** In SQL editor, `select * from pg_policies where schemaname='public';` shows three "owner full access" policies. `select tgname from pg_trigger where tgname='on_auth_user_created';` returns one row.

Create `lib/supabase/server.ts`:
```ts
import { createServerClient } from "@supabase/ssr";
import { cookies } from "next/headers";

export async function createClient() {
  const cookieStore = await cookies();
  return createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() { return cookieStore.getAll(); },
        setAll(cookiesToSet) {
          try {
            cookiesToSet.forEach(({ name, value, options }) =>
              cookieStore.set(name, value, options));
          } catch { /* server component context — ignore */ }
        },
      },
    },
  );
}
```

Create `middleware.ts` at project root to refresh auth tokens (per `@supabase/ssr` Next.js guide):
```ts
import { type NextRequest, NextResponse } from "next/server";
import { createServerClient } from "@supabase/ssr";

export async function middleware(request: NextRequest) {
  let response = NextResponse.next({ request });
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() { return request.cookies.getAll(); },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value }) => request.cookies.set(name, value));
          response = NextResponse.next({ request });
          cookiesToSet.forEach(({ name, value, options }) =>
            response.cookies.set(name, value, options));
        },
      },
    },
  );
  await supabase.auth.getUser();
  return response;
}

export const config = {
  matcher: ["/((?!_next/static|_next/image|favicon.ico).*)"],
};
```

```ts
const supabase = createClient();
async function handleSubmit() {
  if (mode === "register") {
```


Reinstall: `cd backend && source venv/bin/activate && pip install -r requirements.txt`.

Create `backend/app/auth.py`:
```python
import jwt
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPAuthorizationCredentials, HTTPBearer

from app.config import settings

bearer = HTTPBearer()


def current_user_id(creds: HTTPAuthorizationCredentials = Depends(bearer)) -> str:
    try:
        payload = jwt.decode(
            creds.credentials,
            settings.supabase_jwt_secret,
            algorithms=["HS256"],
            audience="authenticated",
        )
    except jwt.PyJWTError as exc:
        raise HTTPException(status.HTTP_401_UNAUTHORIZED, str(exc))
    sub = payload.get("sub")
    if not sub:
        raise HTTPException(status.HTTP_401_UNAUTHORIZED, "missing sub")
    return sub
```

### 5.7 Backend: projects router

Create `backend/app/routers/projects.py`:
```python
from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel

from app.auth import current_user_id
from app.db import supabase

router = APIRouter(prefix="/api/projects", tags=["projects"])


class ProjectIn(BaseModel):
    name: str


class ProjectOut(BaseModel):
    id: str
    name: str


@router.get("", response_model=list[ProjectOut])
def list_projects(user_id: str = Depends(current_user_id)):
    res = supabase().table("projects").select("id,name").eq("user_id", user_id).order("created_at").execute()
    return res.data


@router.post("", response_model=ProjectOut)
def create_project(body: ProjectIn, user_id: str = Depends(current_user_id)):
    res = supabase().table("projects").insert({"user_id": user_id, "name": body.name}).execute()
    return res.data[0]


@router.delete("/{project_id}")
def delete_project(project_id: str, user_id: str = Depends(current_user_id)):
    res = supabase().table("projects").delete().eq("id", project_id).eq("user_id", user_id).execute()
    if not res.data:
        raise HTTPException(404, "not found")
    return {"deleted": project_id}
```

Wire into `app/main.py`:
```python
from app.routers import projects
app.include_router(projects.router)
```

### 5.8 Frontend: hydrate projects after login

In `app-shell.tsx`, after `user` is set, fetch `/api/projects` with the access token (read from `supabase.auth.getSession()`) and replace the fixtures-seeded `projects` state. Keep a small helper `lib/api.ts` that attaches `Authorization: Bearer <access_token>` to every backend call.

`lib/api.ts`:
```ts
import { createClient } from "@/lib/supabase/client";

export async function api(path: string, init: RequestInit = {}) {
  const supabase = createClient();
  const { data: { session } } = await supabase.auth.getSession();
  const headers = new Headers(init.headers);
  if (session) headers.set("Authorization", `Bearer ${session.access_token}`);
  return fetch(`${process.env.NEXT_PUBLIC_API_URL}${path}`, { ...init, headers });
}
```

**Validation:** After signup, sidebar shows the seeded "My Project" (not the two fixtures projects). DevTools network tab shows `Authorization: Bearer ...` on the `/api/projects` call.
